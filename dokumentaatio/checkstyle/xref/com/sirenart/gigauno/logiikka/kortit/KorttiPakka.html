6   package com.sirenart.gigauno.logiikka.kortit;
7   
8   import java.util.ArrayList;
9   import java.util.Collections;
10  import java.util.List;
11  import java.util.Random;
12  
13  /**
14   *
15   * @author arska
16   */
17  public class KorttiPakka {
18  
19      private List<Kortti> kortit;
20  
21      public KorttiPakka(List<Kortti> kortit) {
22          this.kortit = kortit;
23      }
24  
25      public KorttiPakka() {
26          kortit = new ArrayList<>();
27      }
28  
29      public Kortti nosta() {
30          if (kortit.size() > 0) {
31              Kortti nostettava = kortit.get(kortit.size() - 1);
32              kortit.remove(kortit.size() - 1);
33              return nostettava;
34          }
35          return null;
36      }
37  
38      public Kortti ylin() {
39          if (kortit.size() > 0) {
40              Kortti ylinKortti = kortit.get(kortit.size() - 1);
41              return ylinKortti;
42          }
43          return null;
44      }
45  
46      public void lisaa(Kortti kortti) {
47          kortit.add(kortti);
48      }
49  
50      public int jaljella() {
51          return kortit.size();
52      }
53  
54      public void sekoita() {
55          long seed = System.nanoTime();
56          Collections.shuffle(kortit, new Random(seed));
57      }
58  
59      public List<Kortti> getKortit() {
60          return kortit;
61      }
62  
63  }
