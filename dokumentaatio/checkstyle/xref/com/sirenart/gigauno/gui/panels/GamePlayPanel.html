6   package com.sirenart.gigauno.gui.panels;
7   
8   import com.sirenart.gigauno.logiikka.peli.UnoPeli;
9   import java.awt.CardLayout;
10  import java.awt.Dimension;
11  import javax.swing.JPanel;
12  
13  /**
14   * Kontrolleri joka ohjaa pelin kulkua kun peli on käynnissä.
15   *
16   * @author arska
17   */
18  public class GamePlayPanel extends JPanel {
19  
20      private UnoPeli peli;
21  
22      public static final String ERATIEDOT = "ERATIEDOT";
23      public static final String VARMISTUS = "VARMISTUS";
24      public static final String VUORO = "VUORO";
25      public static final String VOITTO = "VOITTO";
26  
27      private VuoroPanel vuoroPanel;
28      private EraTiedotPanel eraTiedotPanel;
29      private VarmistusPanel varmistusPanel;
30      private VoittoPanel voittoPanel;
31  
32      /**
33       * Kontrolleri joka ohjaa pelin kulkua kun peli on käynnissä.
34       *
35       * @param peli Peli jota pelataan.
36       */
37      public GamePlayPanel(UnoPeli peli) {
38          this.peli = peli;
39          peli.aloitaPeli();
40          initComponents();
41          CardLayout cl = (CardLayout) (getLayout());
42          cl.show(this, ERATIEDOT);
43      }
44  
45      private void initComponents() {
46          vuoroPanel = new VuoroPanel(this);
47          eraTiedotPanel = new EraTiedotPanel(this);
48          varmistusPanel = new VarmistusPanel(this);
49          voittoPanel = new VoittoPanel();
50  
51          removeAll();
52          setPreferredSize(new Dimension(800, 800));
53          setLayout(new CardLayout());
54  
55          add(vuoroPanel, VUORO);
56          add(eraTiedotPanel, ERATIEDOT);
57          add(varmistusPanel, VARMISTUS);
58  
59          eraTiedotPanel.paivita(peli);
60  
61          validate();
62  
63      }
64      /**
65       * Kontrolleri metodi silloin kun halutaan aloittaa uusi erä.
66       */
67      public void aloitaUusiEra() {
68  
69          peli.seuraavaEra();
70          CardLayout cl = (CardLayout) (getLayout());
71          varmistusPanel.initializeComponents(peli.getNykyinenEra().nykyinenPelaaja().getNimimerkki());
72          cl.show(this, VARMISTUS);
73      }
74      /**
75       *  Kontrolleri metodi silloin kun vuoro on lopetettu.
76       */
77      public void vuoroOhi() {
78          CardLayout cl = (CardLayout) (getLayout());
79          if (peli.getNykyinenEra().isEraLoppu()) {
80              if (peli.isVoittajaLoytynyt()) {
81                  voittoPanel.initializeComponents(peli.getVoittaja().getNimimerkki());
82                  cl.show(this, VOITTO);
83              }
84              eraTiedotPanel.paivita(peli);
85  
86              validate();
87              cl.show(this, ERATIEDOT);
88          } else {
89              varmistusPanel.initializeComponents(peli.getNykyinenEra().nykyinenPelaaja().getNimimerkki());
90              validate();
91              cl.show(this, VARMISTUS);
92          }
93  
94      }
95      /**
96       *  Kontrolleri metodi silloin kun vuoro on vahvistettu.
97       */
98      public void vuoronVahvistus() {
99          CardLayout cl = (CardLayout) (getLayout());
100         //vuoroPanel
101         validate();
102         vuoroPanel.initComponents();
103         cl.show(this, VUORO);
104     }
105 
106     public UnoPeli getPeli() {
107         return peli;
108     }
109 
110 }
