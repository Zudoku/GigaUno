2   
3   import com.sirenart.gigauno.logiikka.era.PeliAlusta;
4   import com.sirenart.gigauno.logiikka.kortit.Kortti;
5   import com.sirenart.gigauno.logiikka.pelaaja.Pelaaja;
6   import java.awt.Color;
7   import java.awt.Dimension;
8   import java.awt.Font;
9   import java.awt.GridLayout;
10  import java.awt.event.ActionEvent;
11  import java.util.List;
12  import javax.swing.BorderFactory;
13  import javax.swing.JButton;
14  import javax.swing.JLabel;
15  import javax.swing.JPanel;
16  
17  /**
18   * Graafinen käyttöliittymä itse vuoron pelaamiseen. Pelaaja voi laittaa kortin,
19   * nostaa kortin tai huutaa uno tai bustata unon.
20   *
21   * @author arska
22   */
23  public class VuoroPanel extends JPanel {
24  
25      private GamePlayPanel parentPanel;
26      private PeliAlusta alusta;
27  
28      /**
29       * Graafinen käyttöliittymä itse vuoron pelaamiseen. Pelaaja voi laittaa
30       * kortin, nostaa kortin tai huutaa uno tai bustata unon.
31       *
32       * @param parentPanel kontrolleri jota kutsutaan kun ollaan valmiita.
33       */
34      public VuoroPanel(GamePlayPanel parentPanel) {
35          this.parentPanel = parentPanel;
36          initComponents();
37      }
38  
39      /**
40       * Rakentaa käyttöliittymän uusiksi.
41       */
42      public void initComponents() {
43          this.alusta = parentPanel.getPeli().getNykyinenEra();
44          if (alusta == null) {
45              return;
46          }
47          removeAll();
48          invalidate();
49          setPreferredSize(new Dimension(800, 800));
50          setLayout(new GridLayout(3, 3, 20, 20));
51          //Yläbaari
52          add(new JLabel(""));
53          JLabel header = new JLabel("Aika pelata vuorosi!");
54  
55          Font font1 = new Font("SansSerif", Font.BOLD, 15);
56          header.setFont(font1);
57          header.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
58  
59          add(header);
60          add(new JLabel("<html>" + "Kortteja nostopakassa: " + alusta.getNostoPakka().getKortit().size() + "</br>" + " Kortteja laittopakassa: " + alusta.getLaittoPakka().getKortit().size() + "</html>"));
61  
62          //Pelaajien korttimäärä ja uno bustaus napit
63          JPanel players = new JPanel();
64          players.setLayout(new GridLayout(alusta.getPelaajat().size() - 1, 2, 5, 5));
65          for (int i = 0; i < alusta.getPelaajat().size(); i++) {
66              Pelaaja p = alusta.getPelaajat().get(i);
67  
68              if (p != alusta.nykyinenPelaaja()) {
69                  JLabel pelaajaNimiLabel = new JLabel("" + p.getNimimerkki() + " (" + p.getEraTiedot().getKortit().getKortit().size() + ")");
70  
71                  if (p.getEraTiedot().isSaaVoittaa()) {
72                      pelaajaNimiLabel.setBackground(Color.red);
73                  } else if (p.getEraTiedot().onkoPelaajallaUnoTila()) {
74                      pelaajaNimiLabel.setBackground(Color.orange);
75                  }
76                  pelaajaNimiLabel.setBorder(BorderFactory.createEmptyBorder(0, 10, 0, 0));
77                  pelaajaNimiLabel.setOpaque(true);
78                  players.add(pelaajaNimiLabel);
79  
80                  JButton bustaa = new JButton("Paljasta");
81                  bustaa.addActionListener((ActionEvent ae) -> {
82                      alusta.pelajaPaljastaaUnon(p);
83                      initComponents();
84                  });
85                  players.add(bustaa);
86              }
87          }
88          add(players);
89  
90          // Pakan nykyinen kortti
91          Kortti ylin = alusta.getLaittoPakka().ylin();
92          JLabel nykyinen = new JLabel(ylin.getTyyppi().name());
93          nykyinen.setFont(font1);
94          nykyinen.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
95          switch (ylin.getVari()) {
96              case KELTAINEN:
97                  nykyinen.setBackground(Color.yellow);
98                  break;
99  
100             case PUNAINEN:
101                 nykyinen.setBackground(Color.red);
102                 break;
103 
104             case SININEN:
105                 nykyinen.setBackground(Color.blue);
106                 break;
107 
108             case VIHREA:
109                 nykyinen.setBackground(Color.green);
110                 break;
111 
112             case ERIKOIS:
113                 nykyinen.setBackground(Color.DARK_GRAY);
114                 break;
115         }
116         nykyinen.setOpaque(true);
117         add(nykyinen);
118 
119         // Sinun kortit
120         JPanel cards = new JPanel();
121         List<Kortti> kortit = alusta.nykyinenPelaaja().getEraTiedot().getKortit().getKortit();
122         cards.setLayout(new GridLayout(kortit.size(), 2, 5, 5));
123         for (int i = 0; i < kortit.size(); i++) {
124             Kortti p = kortit.get(i);
125             JLabel korttinimi = new JLabel("" + p.getTyyppi().name());
126             korttinimi.setOpaque(true);
127             switch (p.getVari()) {
128                 case KELTAINEN:
129                     korttinimi.setBackground(Color.yellow);
130                     break;
131 
132                 case PUNAINEN:
133                     korttinimi.setBackground(Color.red);
134                     break;
135 
136                 case SININEN:
137                     korttinimi.setBackground(Color.blue);
138                     break;
139 
140                 case VIHREA:
141                     korttinimi.setBackground(Color.green);
142                     break;
143 
144                 case ERIKOIS:
145                     korttinimi.setBackground(Color.DARK_GRAY);
146                     break;
147             }
148 
149             korttinimi.setBorder(BorderFactory.createEmptyBorder(0, 10, 0, 0));
150             cards.add(korttinimi);
151 
152             JButton laita = new JButton("laita");
153             laita.addActionListener((ActionEvent ae) -> {
154                 alusta.pelaajaLaittaaKortin(p);
155                 initComponents();
156             });
157             cards.add(laita);
158         }
159         add(cards);
160         // sano uno
161         JButton sanoUno = new JButton("Sano Uno");
162         if (alusta.nykyinenPelaaja().getEraTiedot().isUnoHuudettu()) {
163             sanoUno.setBackground(Color.green);
164         }
165         sanoUno.addActionListener((ActionEvent ae) -> {
166             alusta.pelaajaHuutaaUno();
167             if (alusta.pelaajaLopettaaVuoron()) {
168                 parentPanel.vuoroOhi();
169             } else {
170                 initComponents();
171             }
172         });
173         add(sanoUno);
174         // Nosta kortti
175         JButton nostaKortti = new JButton("Nosta kortti");
176         nostaKortti.addActionListener((ActionEvent ae) -> {
177             alusta.pelaajaNostaaKortin();
178             if (alusta.pelaajaLopettaaVuoron()) {
179                 parentPanel.vuoroOhi();
180             } else {
181                 initComponents();
182             }
183 
184         });
185         add(nostaKortti);
186         //Lopeta vuoro
187         JButton lopetaVuoro = new JButton("Lopeta");
188         if (!alusta.isPelaajaSaaLopettaaVuoron()) {
189             lopetaVuoro.setBackground(Color.darkGray);
190         }
191         lopetaVuoro.addActionListener((ActionEvent ae) -> {
192             if (alusta.pelaajaLopettaaVuoron()) {
193                 parentPanel.vuoroOhi();
194             }
195 
196         });
197         add(lopetaVuoro);
198 
199         validate();
200     }
201 
202 }
